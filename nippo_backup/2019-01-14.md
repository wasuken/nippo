```
shuffle' :: [Int] -> IO [Int]
shuffle' [] = return []
shuffle' xs = do
  rs <- randomRIO (0,length xs - 1)
  sf <- shuffle' $ drop 1 xs
  return $ nInsert rs (head xs) sf
```
